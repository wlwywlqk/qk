Expression -> AssignExpression
    | BoolExpression
    | EqualityExpression
    | RelExpression
    | ArithExpression
    | UnaryExpression
    | ParenthesisExpression
    | id
    | num
    | str
    | true
    | false
AssignExpression -> id = Expression
BoolExpression -> Expression || Expression
    | Expression && Expression
EqualityExpression -> Expression == Expression
    |= Expression != Expression
RelExpression -> Expression < Expression
    |= Expression <= Expression
    |= Expression > Expression
    |= Expression >= Expression

ArithExpression -> Expression + Expression
    |= Expression - Expression
    | Expression * Expression
    |= Expression / Expression
UnaryExpression -> ! Expression    %right #
    |= - Expression       %right #
ParenthesisExpression -> ( Expression )

Statements -> Statements Statement   # if ($1) $1.push($2) else 
    | Îµ
Statement -> DeclarationStatement
    | ExpressionStatement
    | IfStatement
    | WhileStatement
    |= DoWhileStatement
    | BlockStatement

DeclarationStatement -> Type id
    | Type id = Expression
Type -> var
ExpressionStatement -> Expression
IfStatement -> if ( Expression ) BlockStatement
    | if ( Expression ) BlockStatement else BlockStatement
    | if ( Expression ) BlockStatement else IfStatement
WhileStatement -> while ( Expression ) BlockStatement
DoWhileStatement -> do BlockStatement while ( Expression )
BlockStatement -> { Statements }
